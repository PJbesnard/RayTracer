142  double p = 1 - s.shapes[id] -> spec;
143  bool sample_diffuse;
144  Vector R = r.direction.reflect(N);
145  //We generate either a diffuse sample

148  random_direction = random_cos(N);
149  }
150  //Either a specular sample
151  else {
152  sample_diffuse = false;
153  random_direction = random_phong(R, s.shapes[id] -> phong_expo);
154  if ((dot(random_direction, N) < 0) || (dot(random_direction, R) < 0)) return 1;
155  }
156  Ray random_ray(P + 0.001 * N, random_direction);
157  double proba_phong = (s.shapes[id] -> phong_expo + 1) / (2. * M_PI) * std::pow(dot(R, random_direction), s.shapes[id] -> phong_expo);
158  double proba_globale = p * dot(N, random_direction) / (2. * M_PI) + (1. - p) * proba_phong;
159  if (sample_diffuse) *pixel_color = *pixel_color + getPixelColor(random_ray, s, recursion + 1) * s.shapes[id] -> color * dot(N, random_direction) / M_PI / proba_globale;

161  return 0;

207  void setPixelColor(const Scene& s, int i, int j, double fov, double nb_sampling, unsigned char image[1024*1024*3], int H, int W){
208  //Box Muller's method (anti-alliasing) -> PIXEL SAMPLING
209  //created a random number which follows a Gaussian, it allows to send a ray not necessarily in the center
210  double r1 = generator_rand(generator), r2 = generator_rand(generator);
211  double dx = sqrt(-2 * log(r1)) * cos(2 * M_PI * r2), dy = sqrt(-2 * log(r1)) * sin(2 * M_PI * r2);
212  Vector direction(j - W / 2 + 0.5 * dx , i - H / 2 + 0.5 * dy, - W / (2 * tan(fov / 2)));
213  direction.normalize();
214  Ray r(Vector(0, 0, 0), direction);
215  Vector color(0., 0., 0.);
216  ComplexRayTracer colorPixel;
217  for (int i = 0; i < nb_sampling; i++){
218  color = color + (colorPixel.getPixelColor(r, s) / nb_sampling);
219  }
220  image[((H - i - 1) * W + j) * 3 + 0] = std::min(255., std::max(0., std::pow(color[0], 1 / 2.2))); // red
221  image[((H - i - 1) * W + j) * 3 + 1] = std::min(255., std::max(0., std::pow(color[1], 1 / 2.2))); // green
222  image[((H - i - 1) * W + j) * 3 + 2] = std::min(255., std::max(0., std::pow(color[2], 1 / 2.2))); // blue
